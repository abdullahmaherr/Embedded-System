

#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>


bool findPowerTwo(unsigned int num)
{
	return (1 && (num & (num - 1))); /* 	AND with 1 to return 1 or 0 only,	Don't Forget The Fourmla to Find Power of 2  */
}

long countPairs(unsigned int arr_count, int* arr)
{
	unsigned long counter = 0;
	unsigned int i,j,resAnd = 0;

	for(i = 0; i < (arr_count-1); i++)	 /* 	No mean for last one as j will be greater than arr_count  */
	{
		for(j = (i+1); j < arr_count; j++)    /* J = (i+1) to prevent duplicate of Bitwise AND for example (arr[1]&arr[3]) is equvelant (arr[3]&arr[1]) 	 */
		{

			resAnd = (arr[i] & arr[j]);

			if(resAnd == 0)
			{
				continue;

			}else if(!(findPowerTwo(resAnd)))  /* 		Check the 1 as reult of foumla if number is power 2 is 0	 */
			{
				counter++;
			}
		}
	}

	return counter;
}

int main(void)
{
	/* Read Number of Elements From The User */
	unsigned int arr_count;
	scanf("%d",&arr_count);

	/* Read The Elements From The User */
	int* arr = (int*)malloc(arr_count * sizeof(int));
	for(int i = 0; i < arr_count; i++)
	{
		scanf("%d",&arr[i]);
	}

	unsigned long res = countPairs(arr_count,arr);

	printf("%ld",res);

	free(arr);

	return 0;
}
